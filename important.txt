const { scene } = useGLTF("models/car.glb");
useEffect(() => {
  scene.traverse((child) => {
    if (child.isMesh) {
      child.castShadow = true;
      child.receiveShadow = true;
    }
  });
}, [scene]);

{/* Load and display the car model */}
<primitive position={[0, -1, 0]} object={scene} scale={1} />
